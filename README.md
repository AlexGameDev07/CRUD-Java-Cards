CREATE TABLE tbElectricistas(
    ID_Electricista INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Nombre_Electricista VARCHAR2(100) NOT NULL,
    Apellido_Electricista VARCHAR2(100) NOT NULL,
    Edad_Electricista INT CHECK(Edad_Electricista >= 18) NOT NULL,
    Peso_Electricista NUMBER(5,2) CHECK(Peso_Electricista > 000.00) NOT NULL,
    Correo_Electricista VARCHAR2(100) UNIQUE NOT NULL
);
